create table public.pre_calculations (
  id text not null,
  name text not null,
  date timestamp with time zone null default now(),
  status text not null default 'Draft'::text,
  tax_scheme text not null,
  exchange_rate_usd_kzt numeric not null default 450,
  shipping_china_usd numeric not null default 0,
  shipping_karaganda_kzt numeric not null default 0,
  svh_kzt numeric not null default 0,
  broker_kzt numeric not null default 0,
  customs_fees_kzt numeric not null default 0,
  customs_vat_kzt numeric not null default 0,
  total_volume_m3 numeric null default 0,
  total_weight_kg numeric null default 0,
  notes text null,
  updated_at timestamp with time zone null default now(),
  constraint pre_calculations_pkey primary key (id)
) TABLESPACE pg_default;
